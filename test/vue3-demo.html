⚡ Vue 3 Composition API 集成

// 在 Vue 3 项目中使用
import { ref, computed } from 'vue'
import { StringCaseUtils } from 'string-case-utils'

export default {
  setup() {
    const rawData = ref({
      user_name: 'Alice',
      user_email: 'alice@example.com'
    })
    
    // 响应式转换
    const camelCaseData = computed(() => {
      return StringCaseUtils.convertObjectKeys(rawData.value, {
        caseStyle: 'camel'
      })
    })
    
    return {
      rawData,
      camelCaseData
    }
  }
}
